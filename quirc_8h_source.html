109 /* This structure is used to return information about detected QR codes

112 struct quirc_code {
113  /* The four corners of the QR-code, from top left, clockwise */
114  struct quirc_point corners[4];
115 
116  /* The number of cells across in the QR-code. The cell bitmap
117  * is a bitmask giving the actual values of cells. If the cell
118  * at (x, y) is black, then the following bit is set:
119  *
120  * cell_bitmap[i << 3] & (1 << (i & 7))
121  *
122  * where i = (y * size) + x.
123  */
124  int size;
125  uint8_t cell_bitmap[QUIRC_MAX_BITMAP];
126 };
